 * this is a test of the new mechanical keyboard and so far it is holding up
 * perfectly fine. let's see how well all the key will last overtime.
 */

public class Main {
    public static void main(String[] args) {
        System.out.println("Hello world");
    }

    public static int binarySearch(int[] arr, int ele) {
        int first = 0, last = arr.length, mid = (first + last)/2;
        while(first <= last) {
            mid = (first + last)/2;
            if(ele == arr[mid]) {
                return mid;
            }
            else if(ele < arr[mid]) {
                last = mid - 1;
            }
            else {
                first = mid + 1;
            }
        }
        return -1;
    }
}
